def answer_eleven():

    per_cap = answer_one()['Energy Supply per Capita']
    
    pop = (total / per_cap)
    
    ContinentDict  = {'China':'Asia', 'United States':'North America', 'Japan':'Asia', 'United Kingdom':'Europe', 
                      'Russian Federation':'Europe', 'Canada':'North America', 'Germany':'Europe', 'India':'Asia',
                      'France':'Europe', 'South Korea':'Asia', 'Italy':'Europe', 'Spain':'Europe', 'Iran':'Asia',
                      'Australia':'Australia', 'Brazil':'South America'}
    
    
    for group, frame in pop.groupby(ContinentDict):
        #print(group)
        #print(frame)
        if group == 'Asia':
            Asia = frame.agg([len, sum, np.mean, np.std])
        if group == 'Australia':
            Australia = frame.agg([len, sum, np.mean, np.std])
        if group == 'Europe':
            Europe = frame.agg([len, sum, np.mean, np.std])
        if group == 'North America':
            NAmerica = frame.agg([len, sum, np.mean, np.std])
        if group == 'South America':
            SAmerica = frame.agg([len, sum, np.mean, np.std])

    cont_df = pd.concat([Asia, Australia, Europe, NAmerica, SAmerica], axis = 1).T
    cont_df['Continent'] = ['Asia', 'Australia', 'Europe', 'North America', 'South America']
